## Install

```bash
npm install --save react-lists-dnd
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-lists-dnd'
import 'react-lists-dnd/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
